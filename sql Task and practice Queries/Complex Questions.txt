SELECT B.BOOK_CODE, B.BOOK_TITLE, S.SUPPLIER_NAME
FROM LMS_BOOK_DETAILS B
JOIN LMS_SUPPLIERS_DETAILS S ON B.SUPPLIER_ID = S.SUPPLIER_ID
WHERE B.SUPPLIER_ID = (
    SELECT SUPPLIER_ID
    FROM LMS_BOOK_DETAILS
    GROUP BY SUPPLIER_ID
    ORDER BY COUNT(*) DESC
    LIMIT 1
);



#2
SELECT M.MEMBER_ID, M.MEMBER_NAME, (3 - COALESCE(COUNT(BI.BOOK_CODE), 0)) AS REMAININGBOOKS
FROM LMS_MEMBERS M
LEFT JOIN LMS_BOOK_ISSUE BI ON M.MEMBER_ID = BI.MEMBER_ID
GROUP BY M.MEMBER_ID, M.MEMBER_NAME;




#3
SELECT S.SUPPLIER_ID, S.SUPPLIER_NAME
FROM LMS_SUPPLIERS_DETAILS S
WHERE S.SUPPLIER_ID = (
    SELECT SUPPLIER_ID
    FROM LMS_BOOK_DETAILS
    GROUP BY SUPPLIER_ID
    ORDER BY COUNT(*) ASC
    LIMIT 1
);
